class Board{
  
1 -- * Square;
  1 -- * SpecificPiece;
}

class Square{
  0..1 -- 0..1 SpecificPiece;
  int xCoord;
  int yCoord;
}

class SpecificPiece{
  * -- 1 Player;
  * -- 1 PieceType;
}

class Pawn{
  isA SpecificPiece;
}

class Rook{
  isA SpecificPiece;
}

class Knight{
  isA SpecificPiece;
}

class Bishop{
  isA SpecificPiece;
}

class Queen{
  isA SpecificPiece;
}

class King{
  isA SpecificPiece;
}

class Player{
  name;
  colour;
}

class Turn{
  * -- 1 Player;
  * -- * SpecificPiece;
  int turnNumber;
}

class PieceType{
  description;
}

class Game{
  1 -- 0..1 Board;
  * -- 0..2 Player;
  name;
}
